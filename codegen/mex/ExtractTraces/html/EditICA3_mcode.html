1function [ICThresh,IC,NumICA,x,y] = EditICA(NumICA,MinThreshPixels)
 2%UNTITLED4 Summary of this function goes here
 3%   Detailed explanation goes here
 4
 5if (nargin == 1)
 6  MinThreshPixels = 60;
 7end
 8
 9try
10    % Don't do this if we've done it before
11    load ICAoutlines.mat
12catch
13    
14    % load the ICs
15    for i = 1:NumICA % load the ICA .mat file, put it in a data structure
16        filename = ['Obj_',int2str(i),'_1 - IC filter ',int2str(i),'.mat'];
17        load(filename); % loads two things, Index and Object
18        IC{i} = Object(1).Data;
19    end
20    
21    OG_IC = IC;
22    
23    toContinue = 1;
24    GoodICAbin = ones(1,NumICA);
25    while (toContinue ~= 0)
26        % plot all the ICs
27        close all;
28        Curr = 1;
29        WinCurr = 1;
30        while (Curr <= NumICA)
31            if (mod(Curr,49) == 1)
32                figure;
33                WinCurr = 1;
34            end
35            
36            subplot(7,7,WinCurr);imagesc(IC{Curr});title(int2str(Curr));
37            Curr = Curr+1;
38            WinCurr = WinCurr +1;
39        end
40        
41        % ask the user if any are screwy
42        toEdit = input('Do any of these IC look like non-cells? [y/n]','s');
43        if (strcmp(toEdit,'y') == 1)
44            toDelete = '0';
45            while (strcmp(toDelete,'n') == 0)
46                toDelete = input('Gimme the # of the one to delete, or n to redraw -->','s');
47                if (strcmp(toDelete,'n') == 0)
48                    DeleteNum = str2num(toDelete);
49                    GoodICAbin(DeleteNum) = 0;
50                   
51                end
52            end
53            IC = IC(find(GoodICAbin == 1));
54            NumICA = length(IC);
55            GoodICAbin = ones(1,NumICA);
56        else
57            % none are screwy
58            toContinue = 0;
59        end
60    end
61    
62    
63    
64    for i = 1:NumICA
65        [ICThresh(i),x{i},y{i}] = NumContourPeaks(IC{i},100);
66    end
67    
68    for i = 1:NumICA
69        l(i) = length(find(IC{i} > ICThresh(i)));
70    end
71    figure;hist(l,40);xlabel('# pixels in IC');ylabel('# ICs');
72    LengthGoodidx = find(l > MinThreshPixels);
73    
74    IC = IC(LengthGoodidx);
75    x = x(LengthGoodidx);
76    y = y(LengthGoodidx);
77    ICThresh = ICThresh(LengthGoodidx);
78    
79    NumICA = length(IC);
80    
81    
82    save ICAoutlines.mat IC ICThresh x y NumICA OG_IC;
83end
